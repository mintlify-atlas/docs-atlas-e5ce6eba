## CaseInput

The `CaseInput` model represents the complete input data structure for a medical case, including patient information, doctor's case summary, and associated medical documents.

### Fields

<ParamField path="case_id" type="string" required>
  Unique identifier for the medical case
</ParamField>

<ParamField path="patient_data" type="PatientData" required>
  Patient demographic and basic information
  
  <Expandable title="PatientData fields">
    <ResponseField name="name" type="string" required>
      Patient's full name
    </ResponseField>
    
    <ResponseField name="age" type="integer" required>
      Patient's age in years
    </ResponseField>
    
    <ResponseField name="gender" type="string" required>
      Patient's gender
    </ResponseField>
  </Expandable>
</ParamField>

<ParamField path="doctor_case_summary" type="string" required>
  Doctor's initial case summary and clinical observations
</ParamField>

<ParamField path="lab_files" type="List[ProcessedFile]" default="[]">
  List of laboratory test files associated with this case
  
  <Expandable title="ProcessedFile fields">
    <ResponseField name="file_id" type="string" required>
      Unique identifier for the file
    </ResponseField>
    
    <ResponseField name="file_name" type="string" required>
      Original filename
    </ResponseField>
    
    <ResponseField name="file_type" type="string" required>
      MIME type or file extension
    </ResponseField>
    
    <ResponseField name="file_category" type="string" required>
      File category: either "lab" or "radiology"
    </ResponseField>
    
    <ResponseField name="text_data" type="string">
      Extracted text content from the file
    </ResponseField>
    
    <ResponseField name="ai_summary" type="string">
      AI-generated summary of the file contents
    </ResponseField>
  </Expandable>
</ParamField>

<ParamField path="radiology_files" type="List[ProcessedFile]" default="[]">
  List of radiology files (images, reports) associated with this case
  
  <Expandable title="ProcessedFile fields">
    <ResponseField name="file_id" type="string" required>
      Unique identifier for the file
    </ResponseField>
    
    <ResponseField name="file_name" type="string" required>
      Original filename
    </ResponseField>
    
    <ResponseField name="file_type" type="string" required>
      MIME type or file extension
    </ResponseField>
    
    <ResponseField name="file_category" type="string" required>
      File category: either "lab" or "radiology"
    </ResponseField>
    
    <ResponseField name="text_data" type="string">
      Extracted text content from the file
    </ResponseField>
    
    <ResponseField name="ai_summary" type="string">
      AI-generated summary of the file contents
    </ResponseField>
  </Expandable>
</ParamField>

### Example

```json
{
  "case_id": "CASE-2024-001",
  "patient_data": {
    "name": "John Doe",
    "age": 45,
    "gender": "Male"
  },
  "doctor_case_summary": "Patient presents with chest pain and shortness of breath. History of hypertension.",
  "lab_files": [
    {
      "file_id": "LAB-001",
      "file_name": "blood_test_results.pdf",
      "file_type": "application/pdf",
      "file_category": "lab",
      "text_data": "Complete Blood Count: WBC 8.5, RBC 4.8...",
      "ai_summary": "Blood test shows normal CBC values with slightly elevated lipid panel."
    }
  ],
  "radiology_files": [
    {
      "file_id": "RAD-001",
      "file_name": "chest_xray.png",
      "file_type": "image/png",
      "file_category": "radiology",
      "ai_summary": "Chest X-ray shows clear lung fields with no acute findings."
    }
  ]
}
```

## CaseSummary

Comprehensive summary of a medical case generated by AI analysis, combining information from patient data, doctor's notes, lab results, and radiology findings.

### Fields

<ResponseField name="comprehensive_summary" type="string" required>
  Detailed narrative summary of the entire case
</ResponseField>

<ResponseField name="key_findings" type="List[string]" required>
  List of the most important clinical findings from all sources
</ResponseField>

<ResponseField name="patient_context" type="PatientData" required>
  Patient demographic information
  
  <Expandable title="PatientData fields">
    <ResponseField name="name" type="string" required>
      Patient's full name
    </ResponseField>
    
    <ResponseField name="age" type="integer" required>
      Patient's age in years
    </ResponseField>
    
    <ResponseField name="gender" type="string" required>
      Patient's gender
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="doctor_notes" type="string">
  Doctor's clinical observations and notes
</ResponseField>

<ResponseField name="lab_summary" type="string">
  Aggregated summary of all laboratory findings
</ResponseField>

<ResponseField name="radiology_summary" type="string">
  Aggregated summary of all radiology findings
</ResponseField>

<ResponseField name="confidence_score" type="float" required>
  AI confidence score for the summary (0.0 to 1.0)
</ResponseField>

### Example

```json
{
  "comprehensive_summary": "45-year-old male presenting with acute chest pain. Clinical examination and diagnostic tests suggest possible cardiac involvement. Lab results show elevated cardiac markers. Chest X-ray unremarkable.",
  "key_findings": [
    "Elevated troponin levels",
    "History of hypertension",
    "Chest pain radiating to left arm",
    "Normal chest X-ray"
  ],
  "patient_context": {
    "name": "John Doe",
    "age": 45,
    "gender": "Male"
  },
  "doctor_notes": "Patient presents with chest pain and shortness of breath. History of hypertension.",
  "lab_summary": "Blood work shows elevated cardiac markers with troponin at 0.5 ng/mL. Lipid panel shows borderline high cholesterol.",
  "radiology_summary": "Chest X-ray demonstrates clear lung fields with normal cardiac silhouette.",
  "confidence_score": 0.92
}
```
